p5.js Phyllotaxis Visualization


let n = 0; // Counter for points
const goldenAngle = 137.5; // Golden angle in degrees (approx 360 * (1 - 1/phi))
const scale = 4; // Scale factor for spacing

function setup() {
    let canvas = createCanvas(300, 300);
    canvas.parent('phyllotaxis-canvas');
    background(0); // Black background
    frameRate(60); // Control animation speed
}

function draw() {
    // Center the drawing
    translate(width / 2, height / 2);

    // Add a new point each frame
    let angle = radians(n * goldenAngle); // Convert golden angle to radians
    let radius = scale * sqrt(n); // Distance from center increases with sqrt(n)

    // Calculate position using polar coordinates
    let x = radius * cos(angle);
    let y = radius * sin(angle);

    // Draw the point (small circle)
    fill(255); // White points
    noStroke();
    ellipse(x, y, 5, 5); // Small circle for each point

    n++; // Increment point counter

    // Optional: Stop after a certain number of points
    if (n > 800) {
        noLoop();
    }
}
